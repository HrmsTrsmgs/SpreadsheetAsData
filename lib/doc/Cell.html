def value
  @value ||=
    if not @xml
      BlankValue.new
    else
      case @xml.attributes['t']
      when nil
        @xml.elements[1].text.to_f
      when 'b'
        @xml.elements[1].text != '0'
      when 's'
        book.shared_strings[@xml.elements[1].text.to_i].encode(book.encoding)
      end
    end
end
