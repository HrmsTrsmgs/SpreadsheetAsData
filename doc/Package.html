def self.open(file_path)
  package = Package.new(file_path)
  if block_given?
    begin
      yield package
    ensure
      package.close
    end
  else
    package
  end
end

def close    
  @initialized_parts.select(&:changed?).each do |part|
    @archive.replace_buffer(part.part_uri, part.xml_document.to_s)
  end
  @archive.close
end
