def cell(ref)
  @cell_hash[ref] ||=
    if cell_xml(ref)
      @cell_hash[ref] = Cell.new(cell_xml(ref), self)
    elsif ref =~ %r[A-Z]+\d+/
      @cell_hash[ref] = Cell.new(ref, self)
    end
end

def method_missing(method_name, *args)
  value = self.cell_value(method_name)
  if value.nil?
    super
  else
    value
  end
end
